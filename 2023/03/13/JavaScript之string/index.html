JavaScript之String

const str= “hello world!”

语法 返回值 示例 描述
string.charAt(index) String str.charAt(4) // o 返回指定位置的字符
string.charCodeAt(index) Number str.charCodeAt(4) //111 指定字符的 Unicode 编码
string.concat(str1,str2…) String "HELLO".concat("world") // HELLOworld 连接连个或多个字符串
String.fromCharCode(n1,n2…) String String.fromCharCode(72,69) // HE 将 Unicode 编码转换为一个字符串
string.indexOf(searchvalue,start) Number str.indexOf("hello") // 0 查找字符串
string.includes(searchvalue,start) true/false str.includes("hello") // true 判断是否包含指定字符
string.lastIndexOf(searchvalue,start) Number str.lastIndexOf("world") // 6 查找最后出现的位置
string.match(regexp) Array str.match(/l/g) // [ 'l', 'l', 'l' ] 检索字符串内指定字符
string.repeat(count) string str.repeat(2) // hello wroldhello world 字符串复制指定次数
string.replace(searchvalue，newvalue) string str.repalce("hello", "hi") // hi world 替换指定字符
string.search(index) Number/-1 str.search("ll") // 2 查找指定字符
string.slice(start,end) string str.slice(1，5) // ello 提取字符串片段
string.split(separator, limit) Array str.split(" ") // [ 'hello', 'world' ] 分割字符串返回数组
string.startWith(searchvalue, start) true/false str.startWith("l", 2) // true 检测是否以指定字符串开始
string.substr(start, length) string str.substr(2, 3) // llo 截取字符串
string.substring(from, to) string str.substring(2, 3) // l 截取两个下标之间的字符串
string.toLowerCase() string "HELLO".toLowerCase() // hello 把字符串转换为小写
string.toUpperCase() string str.toUpperCase() // l 把字符串转换为大写
string.trim() string - 删除字符串头尾空白字符
string.toLocaleLowerCase() string - 根据本地主机的语言环境把字符串转换为小写
string.toLocaleUpperCase() string - 根据本地主机的语言环境把字符串转换为大写
string.toString() string - 返回一个字符串
